\chapter*{Glossary}

\begin{acronym}[Glossary]
\acro{Vector Tiles}{Packets of geographic data, packaged into predefined roughly-square shaped "tiles" for transfer over the web}
\acro{Mapbox Streets}{Name of Mapbox's vector tile set}
\acro{Mapbox GL}{Clientside rendering engine}
\acro{Mapbox Studio}{Mapbox's map style editor}
\acro{Imposm3}{Tool which allows to import the OSM Planet file into a database}
\acro{Mapnik}{Tool to generate vector tiles}
\acro{MBTiles}{Container format for storing map tiles in a single file}
\acro{GeoJSON}{File format for encoding a variety of geographic data structures}
\acro{Web GL}{Javascript API for the graphics library in browsers}
\acro{Natural Earth}{Public map dataset}
\acro{OpenStreetMapData}{OpenStreetMap-derived dataset}
\acro{OSM Planet file}{ XML file containing all OpenStreetMap data}
\end{acronym}
